

UPDATE CCHECK 
SET [FILENAME]='AUTOCONV',FILE_USER_ID='SYSTEM',FLEXGEN_DATE=GETDATE()
WHERE RTN_STATUS='P';

UPDATE CCHECK 
SET REMINDER_NO=7,REMINDER_DATE=CONVERT(datetime,CONVERT(varchar(12), GETDATE()))
WHERE RTN_STATUS='O' AND  DATEDIFF(MM,VALUE_DATE,GETDATE())>6;

UPDATE CCHECK 
SET REMINDER_NO=DATEDIFF(MM,VALUE_DATE,GETDATE())+1,
REMINDER_DATE=CONVERT(datetime,CONVERT(varchar(12), GETDATE()))
WHERE RTN_STATUS='O' AND  DATEDIFF(MM,VALUE_DATE,GETDATE())<=6;

UPDATE CCHECK
SET REMINDER_NO=NULL,REMINDER_DATE=NULL
FROM CCHECK a INNER JOIN db_citibank.dbo.ccheck b
ON a.SLNO=b.SLNO AND a.D_CODE=b.D_CODE
WHERE b.[status]='O' b.printflag IS NULL ;


INSERT INTO DD_CCHECK_HIST(D_CODE, SLNO, NONZONE_REF, 
INPUT_BY, INPUT_DATETIME, AUTH_BY, AUTH_DATETIME, MOD_NO, RTN_STATUS, [STATUS], IS_AUTH)
SELECT left(ddrefno,5),right(ddrefno,3), nonzoneref, 
'SYSTEM',GETDATE(),'SYSTEM',GETDATE(),1,[status],'L',1 
FROM db_citibank.dbo.normalcollection;



INSERT INTO DD_NCOLLECTION_HIST(D_CODE, SLNO, NONZONE_REF, AMOUNT, 
VAT, COMMISION, POSTAGE_CHARGE, INSTRUMENT_EXP, 
INPUT_BY, INPUT_DATETIME, AUTH_BY, AUTH_DATETIME, MOD_NO, RTN_STATUS, [STATUS], IS_AUTH)
SELECT left(ddrefno,5),right(ddrefno,3),nonzoneref, collectedAmount, 
vat,0,0,0,
'SYSTEM',GETDATE(),'SYSTEM',GETDATE(),1,[status],'L',1 
FROM db_citibank.dbo.normalcollection;
